DROP TABLE IF EXISTS BUTTON;
CREATE TABLE `BUTTON` (
  `NAME` varchar(30) NOT NULL,
  `TEXT` varchar(30) NOT NULL,
  `X` int(3) NOT NULL,
  `Y` int(3) NOT NULL,
  `WIDTH` int(5) NOT NULL,
  `HEIGHT` int(5) NOT NULL,
  `ENABLED` int(1) NOT NULL,
  `ACTIONL` varchar(1500) DEFAULT NULL,
  PRIMARY KEY (`NAME`),
  UNIQUE KEY `TEXT` (`TEXT`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
DROP TABLE IF EXISTS CHECKBOX;
CREATE TABLE `CHECKBOX` (
  `NAME` varchar(30) NOT NULL,
  `X` int(3) NOT NULL,
  `Y` int(3) NOT NULL,
  `WIDTH` int(5) NOT NULL,
  `HEIGHT` int(5) NOT NULL,
  `ENABLED` int(1) NOT NULL,
  `SELECTED` int(1) NOT NULL,
  `TEXT` varchar(30) NOT NULL,
  `BUTTONGROUP` varchar(20) NOT NULL,
  PRIMARY KEY (`NAME`),
  UNIQUE KEY `TEXT` (`TEXT`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
DROP TABLE IF EXISTS COMBOBOX;
CREATE TABLE `COMBOBOX` (
  `NAME` varchar(30) NOT NULL,
  `X` int(3) NOT NULL,

  `WIDTH` int(5) NOT NULL,

) ENGINE=InnoDB DEFAULT CHARSET=latin1;
